def getRandomIntegers(min_value, max_value, num_points):
    column_data = []
    for i in range(0, num_points):
        column_data.append(randint(min_value, max_value))
    return column_data

class ClusterPlots:
    data = pd.DataFrame()
    num_points = 300
   
    
    def generateData(self):
        x1 = getRandomIntegers(30, 180, self.num_points)
        x2 = getRandomIntegers(30, 180, self.num_points)
        x3 = getRandomIntegers(1, 15, self.num_points)
        self.data['x1'] = x1
        self.data['x2'] = x2
        self.data['x3'] = x3
   
    
    
    def plot2D(self):
        
        kmeans = KMeans(n_clusters=4)
        kmeans.fit(self.data[['x1','x2']].values)
        plt.figure(figsize=(10, 10))
        cmap = sns.cubehelix_palette(dark=.3, light=.8, as_cmap=True)
        ax = sns.scatterplot(x="x1", y="x2", data=self.data[['x1','x2']], \
                    hue= kmeans.predict(self.data[['x1','x2']].values), style = kmeans.predict(self.data[['x1','x2']].values) ,\
                              palette="Set2", s=100)
        plt.title('Feature 1 vs Feature 2 Scatter Plot : 4 Clusters')
        plt.show()
        
        kmeans = KMeans(n_clusters=5)
        kmeans.fit(self.data[['x1','x3']].values)
        plt.figure(figsize=(10, 10))
        cmap = sns.cubehelix_palette(dark=.3, light=.8, as_cmap=True)
        ax = sns.scatterplot(x="x1", y="x3", data=self.data[['x1','x3']], \
                    hue= kmeans.predict(self.data[['x1','x2']].values), style = kmeans.predict(self.data[['x1','x3']].values) ,\
                              palette="Set2", s=100)
        plt.title('Feature 1 vs Feature 3 Scatter Plot : 5 Clusters')
        plt.show()

        
    def plot3D(self):
        kmeans = KMeans(n_clusters=4)
        kmeans.fit(self.data.values)
        trace1 = go.Scatter3d(x = self.data['x1'], y = self.data['x2'], z = self.data['x3'], mode='markers', marker=dict(
                size=12,
                color= kmeans.predict(self.data.values),                
                colorscale='Viridis',   
                opacity=0.8
            )
        )
        
        data_to_plot = [trace1]
        layout = go.Layout( autosize=False,
        width=800,
        height=800,
        title = '3d plot ( x1,x2,x3 features ) with 4 clusters',
         scene = dict(
        xaxis = dict(
            title='Feature 1'),
        yaxis = dict(
            title='Feature 2'),
        zaxis = dict(
        title='Feature 3'),),

        )
        
        fig = go.Figure(data=data_to_plot, layout=layout)
        iplot(fig, filename='3d-scatter-colorscale')
        
        
        
        
        
        kmeans = KMeans(n_clusters=5)
        kmeans.fit(self.data.values)
        trace1 = go.Scatter3d(x = self.data['x1'], y = self.data['x2'], z = self.data['x3'], mode='markers', marker=dict(
                size=12,
                color= kmeans.predict(self.data.values),                
                colorscale='Viridis',   
                opacity=0.8
            )
        )
        
        data_to_plot = [trace1]
        layout = go.Layout( autosize=False,
        width=800,
        height=800,
        title = '3d plot ( x1,x2,x3 features ) with 5 clusters',
         scene = dict(
        xaxis = dict(
            title='Feature 1'),
        yaxis = dict(
            title='Feature 2'),
        zaxis = dict(
        title='Feature 3'),),

        )
        
        fig = go.Figure(data=data_to_plot, layout=layout)
        iplot(fig, filename='3d-scatter-colorscale')
